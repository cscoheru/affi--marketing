**技术规格**:
```javascript
// JavaScript SDK 示例
(function(w,d,s,id) {
  var js, fjs = d.getElementsByTagName(s)[0];
  if (d.getElementById(id)) return;
  js = d.createElement(s); js.id = id;
  js.src = 'https://tracker.zenconsult.top/sdk.js';
  fjs.parentNode.insertBefore(js, fjs);
}(window, document, 'script', 'affi-tracker'));

// 使用
AffiTracker.init('experiment_id');
AffiTracker.track('click', { target_url: '...' });
```
